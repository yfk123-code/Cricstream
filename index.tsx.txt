const VideoPlayer = ({ url, type }: { url: string, type: StreamType }) => {
  const videoRef = useRef<HTMLVideoElement>(null);
  const playerContainerRef = useRef<HTMLDivElement>(null);
  const [isPlaying, setIsPlaying] = useState(false);
  const [isMuted, setIsMuted] = useState(false);
  const [levels, setLevels] = useState<any[]>([]);
  const [currentLevel, setCurrentLevel] = useState(-1);
  const [playbackSpeed, setPlaybackSpeed] = useState(1);
  const [showControls, setShowControls] = useState(true);
  const [isShowingAd, setIsShowingAd] = useState(true);
  
  const AD_TOTAL_DURATION = 15;
  const SKIP_AFTER_SECONDS = 5;
  const [adTimeElapsed, setAdTimeElapsed] = useState(0);
  
  const hlsRef = useRef<any>(null);
  const controlsTimeout = useRef<any>(null);

  const AD_URL = "https://test-streams.mux.dev/x36xhzz/x36xhzz.m3u8";

  const loadSource = (sourceUrl: string, isAd: boolean) => {
    if (hlsRef.current) hlsRef.current.destroy();

    if (type === 'm3u8' && videoRef.current) {
      if ((window as any).Hls.isSupported()) {
        const hls = new (window as any).Hls();
        hls.loadSource(sourceUrl);
        hls.attachMedia(videoRef.current);
        hlsRef.current = hls;
        hls.on((window as any).Hls.Events.MANIFEST_PARSED, () => {
          setLevels(hls.levels);
          if (videoRef.current && isPlaying) videoRef.current.play();
        });
        hls.on((window as any).Hls.Events.LEVEL_SWITCHED, (e: any, d: any) => setCurrentLevel(d.level));
      } else if (videoRef.current.canPlayType('application/vnd.apple.mpegurl')) {
        videoRef.current.src = sourceUrl;
      }
    } else if (type === 'mp4' && videoRef.current) {
      videoRef.current.src = sourceUrl;
    }
  };

  useEffect(() => {
    if (type !== 'embed') loadSource(AD_URL, true);
    return () => { if (hlsRef.current) hlsRef.current.destroy(); };
  }, [url, type]);

  useEffect(() => {
    let timer: any;
    if (isShowingAd && isPlaying && adTimeElapsed < AD_TOTAL_DURATION) {
      timer = setInterval(() => setAdTimeElapsed(prev => prev + 1), 1000);
    } else if (adTimeElapsed >= AD_TOTAL_DURATION && isShowingAd) {
      skipAd();
    }
    return () => clearInterval(timer);
  }, [isShowingAd, isPlaying, adTimeElapsed]);

  const skipAd = () => {
    setIsShowingAd(false);
    setAdTimeElapsed(0);
    loadSource(url, false);
  };

  const togglePlay = () => {
    if (videoRef.current) {
      if (videoRef.current.paused) { videoRef.current.play(); setIsPlaying(true); }
      else { videoRef.current.pause(); setIsPlaying(false); }
    }
  };

  if (!url) {
    return (
      <div className="video-container flex flex-col items-center justify-center glass">
        <Tv className="w-16 h-16 text-gray-600 mb-4" />
        <h3 className="text-xl font-bold text-gray-400">No Stream URL</h3>
      </div>
    );
  }

  if (type === 'embed') {
    return (
      <div className="video-container">
        <iframe src={url} allowFullScreen allow="autoplay; encrypted-media"></iframe>
      </div>
    );
  }

  const canSkip = adTimeElapsed >= SKIP_AFTER_SECONDS;
  const timeUntilSkip = SKIP_AFTER_SECONDS - adTimeElapsed;

  return (
    <div ref={playerContainerRef} className="video-container group relative cursor-pointer" onClick={togglePlay}>
      <video ref={videoRef} className="rounded-lg w-full h-full object-contain" onPlay={() => setIsPlaying(true)} onPause={() => setIsPlaying(false)} autoPlay />

      {isShowingAd && (
        <div className="absolute inset-0 z-10 pointer-events-none">
          <div className="absolute top-4 left-4 bg-black/60 px-3 py-1 rounded-md text-[10px] font-bold text-white border border-white/20 flex items-center gap-2">
            <span className="w-2 h-2 bg-yellow-500 rounded-full animate-pulse"></span> ADVERTISEMENT
          </div>
          <div className="absolute bottom-16 right-4 pointer-events-auto">
             <button onClick={(e) => { e.stopPropagation(); if(canSkip) skipAd(); }} disabled={!canSkip}
                className={`flex items-center gap-2 px-6 py-3 rounded-xl font-bold text-sm transition ${!canSkip ? 'bg-black/40 text-gray-400 border border-white/10 cursor-not-allowed' : 'bg-[#00a3ff] text-white shadow-xl shadow-blue-500/20 active:scale-95'}`}>
                {!canSkip ? `Skip in ${timeUntilSkip}s` : <><SkipForward size={18}/> Skip Ad</>}
             </button>
          </div>
          <div className="absolute bottom-4 left-4 bg-black/60 px-2 py-1 rounded text-[9px] text-gray-300">Ad ends in {AD_TOTAL_DURATION - adTimeElapsed}s</div>
        </div>
      )}

      {!isPlaying && (
        <div className="absolute inset-0 flex items-center justify-center bg-black/40">
          <div className="w-20 h-20 bg-[#00a3ff] rounded-full flex items-center justify-center shadow-2xl animate-pulse">
            <Play size={40} className="text-white ml-1" fill="currentColor" />
          </div>
        </div>
      )}
    </div>
  );
};

const LiveChat = ({ matchId }: { matchId: string }) => {
  const [messages, setMessages] = useState<ChatMessage[]>([]);
  const [input, setInput] = useState('');
  const scrollRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    const saved = localStorage.getItem(`chat_${matchId}`);
    if (saved) setMessages(JSON.parse(saved));
    else setMessages([{ id: 'welcome', user: 'System', text: 'Welcome to the live chat!', timestamp: new Date().toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' }), isMe: false }]);

    const interval = setInterval(() => {
      if (Math.random() > 0.7) {
        const newMsg: ChatMessage = { id: Date.now().toString(), user: "CricFan_" + Math.floor(Math.random() * 100), text: "What a game! ðŸ”¥", timestamp: new Date().toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' }), isMe: false };
        setMessages(prev => {
          const updated = [...prev, newMsg].slice(-30);
          localStorage.setItem(`chat_${matchId}`, JSON.stringify(updated));
          return updated;
        });
      }
    }, 5000);
    return () => clearInterval(interval);
  }, [matchId]);

  useEffect(() => { if (scrollRef.current) scrollRef.current.scrollTop = scrollRef.current.scrollHeight; }, [messages]);

  const handleSend = (e: React.FormEvent) => {
    e.preventDefault();
    if (!input.trim()) return;
    const myMsg: ChatMessage = { id: Date.now().toString(), user: 'You', text: input, timestamp: new Date().toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' }), isMe: true };
    const updated = [...messages, myMsg].slice(-30);
    setMessages(updated);
    localStorage.setItem(`chat_${matchId}`, JSON.stringify(updated));
    setInput('');
  };

  return (
    <div className="glass rounded-2xl flex flex-col h-[350px] border border-gray-800 overflow-hidden mt-4">
      <div className="p-4 border-b border-gray-800 bg-white/5 flex items-center justify-between">
        <h3 className="text-sm font-bold flex items-center gap-2 text-[#00a3ff]"><MessageSquare size={16} /> LIVE CHAT</h3>
      </div>
      <div ref={scrollRef} className="flex-1 overflow-y-auto p-4 space-y-4 hide-scrollbar">
        {messages.map((msg) => (
          <div key={msg.id} className={`flex flex-col ${msg.isMe ? 'items-end' : 'items-start'}`}>
            <span className="text-[10px] text-gray-500 mb-1">{msg.user} â€¢ {msg.timestamp}</span>
            <div className={`px-3 py-2 rounded-xl text-sm max-w-[85%] ${msg.isMe ? 'bg-[#00a3ff] text-white rounded-tr-none' : 'bg-gray-800 text-gray-200 rounded-tl-none'}`}>{msg.text}</div>
          </div>
        ))}
      </div>
      <form onSubmit={handleSend} className="p-3 border-t border-gray-800 bg-black/20 flex gap-2">
        <input type="text" value={input} onChange={(e) => setInput(e.target.value)} placeholder="Say something..." className="flex-1 bg-white/5 border border-gray-800 rounded-lg px-3 py-2 text-sm outline-none focus:border-[#00a3ff]" />
        <button type="submit" className="bg-[#00a3ff] p-2 rounded-lg text-white"><Send size={18} /></button>
      </form>
    </div>
  );
};

                  <thead><tr className="bg-white/5 text-[10px] font-black uppercase text-gray-500 border-b border-gray-800"><th className="px-8 py-5">Match Details</th><th className="px-8 py-5">Tournament</th><th className="px-8 py-5 text-right">Delete</th></tr></thead>
                  <tbody className="divide-y divide-gray-800">
                    {matches.map(m => (
                      <tr key={m.id} className="hover:bg-white/5 transition"><td className="px-8 py-6"><p className="font-black uppercase tracking-tight">{m.title}</p><p className="text-[10px] text-gray-500">{m.date} - {m.time}</p></td><td className="px-8 py-6 text-sm font-bold text-[#00a3ff] uppercase">{m.tournament}</td><td className="px-8 py-6 text-right"><button onClick={() => deleteMatch(m.id)} className="p-3 text-red-500 hover:bg-red-500/10 rounded-xl"><Trash2 size={20}/></button></td></tr>
                    ))}
                  </tbody>
               </table>
            </div>
          </div>
        )}
      </div>
      <footer className="text-center py-10 opacity-20 text-[10px] font-black uppercase tracking-[0.3em] mt-20 border-t border-gray-900">
        &copy; 2026 CRICSTREAM NETWORK PRO
      </footer>
    </div>
  );
};
